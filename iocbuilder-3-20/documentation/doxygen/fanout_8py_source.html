00016 # Support routine to do the work of fanout generation common to fanout and
00017 # dfanout.
00018 def _fanout_helper(
00019     fanout_name, link_list, fanout_size,
00020     record_factory, field_name, fixup_link, firstargs, nextargs):
00021 
00022     # First break the list of links into chunks small enough for each fanout
00023     # record.  First chop it into segments small enough to fit into each
00024     # fanout record, leaving room for an extra link.  The last record can take
00025     # an extra entry so then fix up the chopped list.
00026     chopped = support.choplist(link_list, fanout_size - 1)
00027     if len(chopped) > 1 and len(chopped[-1]) == 1:
00028         chopped[-2:] = [chopped[-2] + chopped[-1]]
00029 
00030     # Convert the chopped list into a list of fanout records.
00031     recordList = []
00032     args = firstargs
00033     for i, links in enumerate(chopped):
00034         # The first record gets the standard name and a different set of
00035         # record arguments.
00036         name = fanout_name
00037         if i > 0:
00038             name += str(i)
00039         # Build a fanout record with the computed name and arguments.
00040         record = record_factory(name, **args)
00041         args = nextargs     # Subsequent records get the other arguments
00042         # Link the new fanout record to the given list of links
00043         for i, link in enumerate(links):
00044             setattr(record, field_name(i), link)
00045         recordList.append(record)
00046 
00047     # Chain the fanout records together using the last field in each record:
00048     # we've taken care to reserve this field when we split the link list!
00049     next_name = field_name(fanout_size - 1)
00050     for prev, next in zip(recordList[:-1], recordList[1:]):
00051         setattr(prev, next_name, fixup_link(next))
00052 
00053     return recordList

00057 ## Creates one or more fanout records (as necessary) to fan processing out to
00058 # a list of records.  If no more than 6 records are given then this creates a
00059 # single fanout record called \c name.  Otherwise, further records are created
00060 # and daisy-chained together: subsequent records are numbered in sequence.
00061 #
00062 # \param name
00063 #   Name of first fanout record.  Subsequent records have a sequence number
00064 #   appended to this name.
00065 # \param *record_list
00066 #   List of records to be processed.
00067 # \param **args
00068 #   Extra field definitions to be passed to the generated fanout records.
00069 def create_fanout(name, *record_list, **args):
00070     # We can only support fanout to "All" style fanout records: to generate
00071     # masked or selected fanouts we'd need to create a cluster of supporting
00072     # calc records and structure the set rather differently.
00073     args['SELM'] = 'All'
00074 
00075     # All records after the first must be passive.
00076     firstargs = args
00077     nextargs = args.copy()
00078     nextargs['SCAN'] = 'Passive'
00079     if nextargs.has_key('PINI'):  del nextargs['PINI']
00080 
00081     def fieldname(i):   return 'LNK%d' % (i + 1)
00082     def identity(x):    return x
00083     record_list = _fanout_helper(
00084         name, record_list, 6, records.fanout, fieldname,
00085         identity, firstargs, nextargs)
00086     return record_list[0]
