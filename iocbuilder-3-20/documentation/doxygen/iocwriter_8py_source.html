00082     def write(self, string):
00083         # Check that no line exceeds the maximum line length
00084         lines = string.split('\n')
00085         for line in lines[:-1]:
00086             full_line = self.__line + line
00087             self.__line = ''
00088             assert len(full_line) <= self.__maxLineLength, \
00089                 'Line %s too long' % repr(full_line)
00090         self.__line = self.__line + lines[-1]
00091 
00092         self.__output.write(string)

00094     # Call this to close the file being written and to restore normal output
00095     # to stdout.
00096     def Close(self):
00097         assert self.__output != None, 'Close called out of sequence.'
00098         assert len(self.__line) <= self.__maxLineLength, \
00099             'Unterminated line %s too long' % repr(self.__line)
00100         self.__output.close()

00466         # Checks that the newly computed iocBoot directory is a plausible IOC
