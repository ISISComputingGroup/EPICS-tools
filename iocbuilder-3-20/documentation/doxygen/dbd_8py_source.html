00072     # Computes list of valid names and creates associated arginfo
00073     # definitions.  This is postponed quite late to try and ensure the menus
00074     # are fully populated, in other words we don't want to fire this until
00075     # all the dbd files have been loaded.
00076     def __ProcessDbd(self):
00077         # ordered dict of field_name -> arginfo
00078         self._FieldInfo = OrderedDict()
00079         valid_names = []
00080         status = mydbstatic.dbFirstField(self.dbEntry, 0)
00081         while status == 0:
00082             name = mydbstatic.dbGetFieldName(self.dbEntry)
00083             desc = mydbstatic.dbGetPrompt(self.dbEntry)
00084             typ = mydbstatic.dbGetFieldType(self.dbEntry)
00085             group = mydbstatic.dbGetPromptGroup(self.dbEntry)
00086             if typ in [DCT_STRING, DCT_INLINK, DCT_OUTLINK, DCT_FWDLINK]:
00087                 ArgInfo = arginfo.Simple(desc, str)
00088             elif typ in [DCT_INTEGER]:
00089                 ArgInfo = arginfo.Simple(desc, int)
00090             elif typ in [DCT_REAL]:
00091                 ArgInfo = arginfo.Simple(desc, float)
00092             elif typ in [DCT_MENU, DCT_MENUFORM]:
00093                 n_choices = mydbstatic.dbGetNMenuChoices(self.dbEntry)
00094                 if n_choices > 0:
00095                     menu_void = mydbstatic.dbGetMenuChoices(self.dbEntry)
00096                     menu_p = ctypes.cast(menu_void,
00097                         ctypes.POINTER(ctypes.c_char_p * n_choices))
00098                     ArgInfo = arginfo.Choice(desc, list(menu_p[0]))
00099                 else:
00100                     ArgInfo = arginfo.Simple(desc, str)
00101             else:
00102                 # No access field.
00103                 ArgInfo = None
00104             if name != "NAME":
00105                 valid_names.append(name)
00106                 if ArgInfo is not None:
00107                     self._FieldInfo[name] = ArgInfo
00108             status = mydbstatic.dbNextField(self.dbEntry, 0)
00109 
00110         self._ValidNamesSet = set(valid_names)
