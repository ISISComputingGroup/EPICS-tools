00001 '''Support for hardware devices.  A device is supported in three ways:
00002     1. when a device is used its supporting library must be loaded into

00004     2. when a device is used the device itself must be initialised;

00206         # invoked at phase 0.

00350     # Calling a record factory instance builds a record with the given name
00351     # and fields bound to the originating hardware.
00352     def __call__(self, name, *address_extra, **fields):
00353         # If the address is callable then we compute the address using the
00354         # address hook, simultaneously fixing up any fields that only belong
00355         # to the address computation.  Otherwise we hope it's a static string
00356         # (or at least knows how to render itself) and use it as is.
00357         if callable(self.address):
00358             address = self.address(fields, *address_extra)
00359         else:
00360             assert address_extra == (), 'Unused address arguments'
00361             address = self.address
00362         # Build the appropriate type of record
00363         record = self.factory(name, **fields)

00366         record.DTYP = self.device
00367         setattr(record, self.link, address)

00369         # Finally allow any special purpose fixup work to be done.
