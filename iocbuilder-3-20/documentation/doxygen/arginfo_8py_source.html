00032     #     .descriptions       Dictionary mapping argument names to meta-data
00033     #     .required_names     List of mandatory arguments (without defaults)

00038     def __init__(self, __source=None, __optional=[], __method=True, **descs):
00039         self.descriptions = descs
00040         for k, v in descs.items():
00041             assert isinstance(v, ArgType), \
00042                 'ArgInfo description "%s" is not of valid type. Got:\n%s' % \
00043                     (k, v)
00044         self.optional_names = list(__optional)
00045 
00046         if callable(__source):
00047             # First get the names and defaults from the given function
00048             names, _, varkw, defaults = inspect.getargspec(__source)
00049             # If this is a bound method, discard the first argument.
00050             if __method:
00051                 names = names[1:]
00052             if defaults:
00053                 self.required_names = names[:-len(defaults)]
00054                 self.default_names = names[-len(defaults):]
00055                 self.default_values = list(defaults)
00056             else:
00057                 self.required_names = names
00058                 self.default_names = []
00059                 self.default_values = []
00060 
00061             if varkw:
00062                 # Add any names in descs that we've not already accounted for
00063                 # to the list of optional names.  We sort the list so that
00064                 # the ordering is at least predictable.
00065                 self.required_names += sorted(list(
00066                     set(descs) - set(names) - set(__optional)))
00067             assert varkw or not __optional, \
00068                 'Unusable optional arguments: %s' % self.optional_names
00069         else:
00070             # Somewhat special case for taking the arguments from a list of
00071             # names.  These are treated as all mandatory.
00072             if __source:
00073                 self.required_names = list(__source)
00074             else:
00075                 self.required_names = sorted(set(descs) - set(__optional))
00076             self.default_names = []
00077             self.default_values = []

00082     def __validate(self):
00083         # Validates the complete ArgInfo data.
00084         set_required = set(self.required_names)
00085         set_default  = set(self.default_names)
00086         set_optional = set(self.optional_names)
00087 
00088         # First check for duplicates in all the lists of names.
00089         assert len(set_optional) == len(self.optional_names), \
00090             'Duplicate optional names'
00091         assert len(set_required) == len(self.required_names), \
00092             'Duplicate required names'
00093         assert len(set_default) == len(self.default_names), \
00094             'Duplicate default names'
00095         # Check for overlaps between the lists
00096         assert \
00097             not (set_optional & set_required) and \
00098             not (set_required & set_default)  and \
00099             not (set_optional & set_default), 'Overlapping arguments?'
00100 
00101         # Check that all the names are described
00102         all_names = set_required | set_default | set_optional
00103         set_descs = set(self.descriptions)
00104         assert all_names <= set_descs, \
00105             'Arguments %s not described' % (all_names - set_descs)
00106         assert set_descs <= all_names, \
00107             'Descriptions for unknown arguments %s' % (set_descs - all_names)

00118     # object.  Argument details in the second ArgInfo object override settings
00119     # in the first.
00120     def __add__(self, other):
00121         result = ArgInfo()
00122         result.descriptions = dict(self.descriptions, **other.descriptions)
00123         result.required_names = self.required_names[:]
00124         result.default_names  = self.default_names[:]
00125         result.default_values = self.default_values[:]
00126         result.optional_names = self.optional_names[:]
00127 
00128         # Because any type of argument can be convered to any other type, but
00129         # we are keen to preserve argument order, the process of merging is
00130         # rather involved.
00131         def del_required(name):
00132             if name in result.required_names:
00133                 result.required_names.remove(name)
00134         def del_optional(name):
00135             if name in result.optional_names:
00136                 result.optional_names.remove(name)
00137         def del_default(name):
00138             if name in result.default_names:
00139                 i = result.default_names.index(name)
00140                 del result.default_names[i]
00141                 del result.default_values[i]
00142 
00143         for name in other.required_names:
00144             del_optional(name)
00145             del_default(name)
00146             if name not in result.required_names:
00147                 result.required_names.append(name)
00148         for (name, value) in zip(other.default_names, other.default_values):
00149             del_required(name)
00150             del_optional(name)
00151             if name in result.default_names:
00152                 result.default_values[result.default_names.index(name)] = value
00153             else:
00154                 result.default_names.append(name)
00155                 result.default_values.append(value)
00156         for name in other.optional_names:
00157             del_required(name)
00158             del_default(name)
00159             if name not in result.optional_names:
00160                 result.optional_names.append(name)

00162         result.__validate()

00228 ## A choice of different values, with optional different labels
00229 # \param desc Description of the argument
00230 # \param values List of possible values, type must be one of _simpleTypes, and
00231 # the same for all in the list of values
00232 # \param labels List of optional labels for the values
00233 def Choice(desc, values, labels = None):
00234     typ = type(values[0])
00235     assert typ in _simpleTypes, \
00236         '%s is not a supported simple type %s' % (typ, _simpleTypes)
00237     for v in values:
00238         assert typ == type(v), \
00239             'Value "%s" doesn\'t have same type as "%s"' % (v, values[0])
00240     desc = '%s\n%s\nValues:\n ' % (desc, typ)
00241     if labels is None:
00242         desc += '\n '.join(map(str, values))
00243         return ArgType(desc, typ, labels = values)
00244     else:
00245         assert len(values) == len(labels), \
00246             'Labels %s do not have the same length as values %s' % \
00247             (labels, values)
00248         argdescs = [ '%s: %s' % x for x in zip(labels, values) ]
00249         desc += '\n '.join(argdescs)
00250         return ArgType(desc, typ, values = values, labels = labels)
