00029 def ValidateEpicsBase(epics_base):
00030     '''Very DLS specific validation and normalisation of EPICS_BASE.'''
00031     path = epics_base.split('/')
00032     if path[-1] == '':
00033         del path[-1]
00034     assert len(path) == 5 and \
00035         '/'.join(path[:3]) == '/dls_sw/epics' and \
00036         path[4] == 'base', 'Invalid DLS EPICS_BASE: "%s"' % epics_base
00037     return '/'.join(path)

00039 def _GetEpicsVersion(epics_base):
00040     '''Very DLS specific extraction of EPICS version from epics_base.'''
00041     epics_base = epics_base.split('/')
00042     # Try to recognise standard form:
00043     #   /dls_sw/epics/$EPICS_RELEASE/base
00044     if epics_base[-1] == '':
00045         del epics_base[-1]
00046     if '/'.join(epics_base[:3] + epics_base[4:]) == '/dls_sw/epics/base':
00047         return epics_base[3]
00048     else:
00049         return None
