00253 def QuoteArgument(argument):

00255     #    http://www.aps.anl.gov/asd/controls/epics/EpicsDocumentation/
00256     #        ExtensionsManuals/msi/msi.html
00257     # it is enough to quote backquotes (and presumably backslashes).

00259         if char in '"\\':

00262             return char
