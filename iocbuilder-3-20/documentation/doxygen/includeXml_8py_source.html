00099 def constructArgDict(el, objects, classes, ident_lookup=True):
00100     # dict of args to return
00101     d = {}
00102     # name of object represented by this element
00103     name = None
00104     # get the object class
00105     obname = str(el.nodeName)
00106     assert obname in classes, 'Can\'t find object "%s"'% obname
00107     ob = classes[obname]
00108     # find the column representing name
00109     nameKey = getattr(ob, 'UniqueName', 'name')
00110     # see if this nameKey also needs to be passed to the object constructor
00111     needsNameKey = nameKey in ob.ArgInfo.descriptions
00112     # build up the arg dict
00113     for attr, value in el.attributes.items():
00114         attr = str(attr)
00115         value = str(value)
00116         # check if this is the name key
00117         if attr == nameKey:
00118             name = value
00119             # if we don't need the name, then continue
00120             if not needsNameKey:
00121                 continue
00122         # check the attribute is a valid argument
00123         assert attr in ob.ArgInfo.descriptions, \
00124             '%s is not a valid argument for %s' %(attr, ob)
00125         desc = ob.ArgInfo.descriptions[attr]
00126         # if it is an ident look it up
00127         if getattr(desc, 'ident', False):
00128             if ident_lookup:
00129                 assert value in objects, 'Can\'t find object %s' % value
00130                 value = objects[value]
00131         # otherwise make it the right type
00132         elif desc.typ == bool:
00133             if value and value.lower() != 'false':
00134                 value = True
00135             else:
00136                 value = False
00137         elif desc.typ == str:
00138             value = value.decode('string_escape')
00139         else:
00140             value = desc.typ(value)
00141         # add it to the dict
00142         d[attr] = value
00143     return name, ob, d
