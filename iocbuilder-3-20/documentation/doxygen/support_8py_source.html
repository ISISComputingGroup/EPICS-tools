00131 # simply return the original class.  This behaviour is pretty transparent.
00132 #
00133 # The role of this Singleton class is a little unclear.  It can readily be

00172 # Meta-class to implement __super attribute in all subclasses.  To use this
00173 # define the metaclass of the appropriate base class to be autosuper thus:
00174 #
00175 #     class A:
00176 #         __metaclass__ = _autosuper_meta
00177 #
00178 # Then in any sub-class of A the __super attribute can be used instead of
00179 # writing super(cls, name) thus:
00180 #
00181 #     class B(A):
00182 #         def __init__(self):
00183 #             self.__super.__init__()
00184 #             # instead of
00185 #             # super(B, self).__init__()
00186 #
00187 # The point being, of course, that simply writing
00188 #             A.__init__(self)
00189 # will not properly interact with calling order in the presence of multiple
00190 # inheritance: it may be necessary to call a sibling of B instead of A at
00191 # this point!
00192 #
00193 # Note that this trick does not work properly if a) the same class name
00194 # appears more than once in the class hierarchy, and b) if the class name is
00195 # changed after it has been constructed.  The class can be renamed during
00196 # construction by setting the TrueName attribute.
00197 #
00198 #
00199 # This meta class also supports the __init_meta__ method: if this method is
00200 # present in the dictionary of the class it will be called after completing
00201 # initialisation of the class.
00202 #
00203 # For any class with _autosuper_meta as metaclass if a method
00204 #     __init_meta__(cls, subclass)
00205 # is defined then it will be called when the class is declared (with subclass
00206 # set to False) and will be called every time the class is subclassed with
00207 # subclass set to True.
00208 #
00209 # This class definition is taken from
00210 #   http://www.python.org/2.2.3/descrintro.html#metaclass_examples
00211 class _autosuper_meta(type):
00212 
00213     __super = property(lambda subcls: super(_autosuper_meta, subcls))

00215     def __init__(cls, name, bases, dict):
00216         # Support renaming.
00217         if 'TrueName' in dict:
00218             name = dict['TrueName']
00219             cls.__name__ = name

00221         cls.__super.__init__(name, bases, dict)
00222

00224         assert not hasattr(cls, super_name), \
00225             'Can\'t set super_name on class %s, name conflict' % name
00226         setattr(cls, super_name, super(cls))

00228         # Support __init_meta__
00229         cls.__call_init_meta(set([object]), cls, False)

00231     def __call_init_meta(cls, visited, base, subclass):
00232         # This is rather painful.  Ideally each __init_meta__ call would
00233         # invoke the appropriate superclass ... unfortunately it can't call
00234         # super() because the class name isn't available yet!
00235         #    So instead we walk the entire class tree, calling __init_meta__
00236         # on all base classes ourself.
00237         for b in base.__bases__:
00238             if b not in visited:
00239                 visited.add(b)
00240                 cls.__call_init_meta(visited, b, True)
