00056 # Factory for building bits from an existing record.  The parent record
00057 # (generally assumed to be an mbbi record) should present the bits to be
00058 # read or written in fields B0 to BF.
00059 #
00060 # Depending on direction, bi or bo can be invoked to create records linking
00061 # to the parent record.  Also bit and register can be used to postpone
00062 # the creation of real records!

00065         assert 0 <= offset  and  offset + length <= 16, \
00066             'Bit field out of range'

00068         self.direction = direction
00069         self.offset = offset
00070         self.length = length
00071         self._record = factory
00072         if direction == BIT_INPUT:
00073              self.bi = self._makeBit
00074              self._field = 'INP'
00075              self._linkage = 'CP'
00076         if direction == BIT_OUTPUT:
00077              self.bo = self._makeBit
00078              self._field = 'OUT'
00079              self._linkage = 'PP'

00082     def _makeBit(self, record, bit=0, **fields):
00083         assert 0 <= bit  and  bit < self.length, 'Bit out of range'
00084         r = self._record(record, **fields)

00086             getattr(self.parent, 'B%X' % (self.offset + bit))(self._linkage))
00087         return r
